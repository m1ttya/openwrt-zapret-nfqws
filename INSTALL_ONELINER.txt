# One-line online installer (basic)
wget -qO- "https://raw.githubusercontent.com/m1ttya/openwrt-zapret-nfqws/main/openwrt/online_install.sh" | sh

# One-line installer using NFQWS IPK (Keenetic mips-3.4 example)
